  1. import java.io.*;
  2. import java.util.Scanner;
  3. public class b{
  4. public static void main(String args[]){
  5. System.out.println("请输入x坐标,若为负数记得先-1,例如-1变-2");
  6. Scanner reader=new Scanner(System.in);
  7. int cx = reader.nextInt();
  8. System.out.println("请输入z坐标,若为负数记得先-1,例如-1变-2");
  9. int cz = reader.nextInt();
  10. int ccx = cx/16;
  11. int ccz = cz/16;
  12. System.out.println("区块x坐标为："+ccx);
  13. System.out.println("区块z坐标为："+ccz);
  14. int rx = ccx >> 5;
  15. int rz = ccz >> 5;
  16. int irx = ccx<0 ? 31-Math.abs(ccx%32):ccx%32;
  17. int irz = ccz<0 ? 31-Math.abs(ccz%32):ccz%32;
  18. System.out.println("方块位于r."+rx+"."+rz+".mca"+"里面的["+irx+","+irz+"]");
  19. }
  20. }
